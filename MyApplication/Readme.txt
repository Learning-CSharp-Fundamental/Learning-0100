--------------------------------------------------
 • نمایش متن، در محیط کنسول اپلیکیشن
--------------------------------------------------

-------------------------
 • System.Console.Write("Hello, World!");
-------------------------

 - مکان نما در همان خط میماند

-------------------------

-------------------------
 • System.Console.WriteLine("Hello World!");
-------------------------

 - مکان نما به خط بعد میرود

-------------------------

-------------------------
 • System.Console.ReadKey();
-------------------------

 - با فشردن هر دکمه ای پنجره ی کنسول بسته می شود

-------------------------

-------------------------
 • System.Console.ReadLine();
-------------------------

 - پنجره ی کنسول بسته می شود ،Enter با فشردن دکمه ی

-------------------------

--------------------------------------------------

--------------------------------------------------
 • Graphic Design in Console Application:
--------------------------------------------------

-------------------------
 - یا عنوان Title تغییر
-------------------------

 System.Console.Title =
	"Graphics in Console Application!";

-------------------------

-------------------------
 - تغییر عرض پنجره ی کنسول:
-------------------------

 System.Console.WindowWidth = 50;

 → !این دستور، صرفا و فقط در محیط ویندوز کارایی دارند

-------------------------

-------------------------
 - تغییر ارتفاع پنجره ی کنسول:
-------------------------

 System.Console.WindowHeight = 10;

 → !این دستور، صرفا و فقط در محیط ویندوز کارایی دارند

 -------------------------

-------------------------
 - پاک کردن محیط کنسول
-------------------------

 System.Console.Clear();

-------------------------

-------------------------
 - تغییر رنگ پس زمینه
-------------------------

 System.Console.BackgroundColor = System.ConsoleColor.White;

 -------------------------

-------------------------
 - تغییر رنگ فونت
-------------------------

 System.Console.ForegroundColor = System.ConsoleColor.Red;

-------------------------
